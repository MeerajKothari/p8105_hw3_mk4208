``` r
instacart %>% 
  group_by(aisle) %>% 
  summarise(n = n()) %>% 
  filter(n > 10000) %>% 
  arrange(desc(n)) %>% 
  ggplot(aes(x = reorder(aisle, n), y = n)) +
  geom_bar(stat = "identity") + 
  scale_y_continuous(
    breaks=seq(0, 160000, 10000)
    ) +
  coord_flip() + 
  theme(axis.text.x = element_text(angle = 45, hjust = 1))
```
